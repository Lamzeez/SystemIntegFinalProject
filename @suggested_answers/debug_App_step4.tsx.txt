import 'react-native-gesture-handler';
import React from 'react';
import { View, Text, StyleSheet, ActivityIndicator } from 'react-native';
import { AuthProvider, useAuth } from './src/AuthContext';
import { NavigationContainer } from '@react-navigation/native';
import { createStackNavigator } from '@react-navigation/stack';
import { AuthStackParamList } from './src/navigation/types';

// Create the stack navigator
const AuthStack = createStackNavigator<AuthStackParamList>();

// A temporary, simple component to use as a screen
function SimpleLoginScreen() {
  return (
    <View style={{ flex: 1, justifyContent: 'center', alignItems: 'center', backgroundColor: 'pink' }}>
      <Text style={{ fontSize: 24, color: 'black', textAlign: 'center' }}>
        This is a test screen inside the navigator. Does it render?
      </Text>
    </View>
  );
}

// A simplified navigator that ONLY shows our simple screen
function AuthStackScreens() {
  return (
    <AuthStack.Navigator>
      <AuthStack.Screen
        name="Login"
        component={SimpleLoginScreen}
        options={{ title: 'Test Screen' }} // Add a title to be sure
      />
    </AuthStack.Navigator>
  );
}

// The navigator that decides what to show
function RootNavigator() {
  const { token, isLoading } = useAuth();

  if (isLoading) {
    // Show a loading indicator while we check for a token
    return (
      <View style={{ flex: 1, justifyContent: 'center', alignItems: 'center' }}>
        <ActivityIndicator size="large" />
      </View>
    );
  }

  // For this test, we ignore the 'token' and always show the auth stack.
  // We are not testing the AppStack (HomeScreen) yet.
  if (token) {
    // In a real scenario this would show the AppStack, but we're debugging
    return <AuthStackScreens />;
  } else {
    return <AuthStackScreens />;
  }
}

export default function App() {
  return (
    <AuthProvider>
      <NavigationContainer>
        <RootNavigator />
      </NavigationContainer>
    </AuthProvider>
  );
}
