import 'react-native-gesture-handler';
import React from 'react';
import { View, Text, StyleSheet, ActivityIndicator } from 'react-native';
import { AuthProvider, useAuth } from './src/AuthContext';
import { NavigationContainer } from '@react-navigation/native';

function AppNavigator() {
  const { token, isLoading } = useAuth();

  // --- This is the new logic we are testing ---
  if (isLoading) {
    return (
      <View style={styles.loadingContainer}>
        <ActivityIndicator size="large" color="#0000ff" />
        <Text style={{ marginTop: 10 }}>Loading...</Text>
      </View>
    );
  }

  // If not loading, show this screen
  return (
    <View style={styles.container}>
      <Text style={styles.text}>Finished loading!</Text>
      <Text style={styles.text}>Token: {token ? 'Present' : 'null'}</Text>
    </View>
  );
}

export default function App() {
  return (
    <AuthProvider>
      <NavigationContainer>
        <AppNavigator />
      </NavigationContainer>
    </AuthProvider>
  );
}

const styles = StyleSheet.create({
  // The main container for after loading
  container: {
    flex: 1,
    justifyContent: 'center',
    alignItems: 'center',
    backgroundColor: '#FF6347', // "Tomato" color
  },
  // A separate container for the loading state
  loadingContainer: {
    flex: 1,
    justifyContent: 'center',
    alignItems: 'center',
  },
  text: {
    fontSize: 20,
    color: 'white',
    fontWeight: 'bold',
    textAlign: 'center',
    margin: 5,
  },
});
